# this command is used to generate reels from photos and audios through ffmpeg#
ffmpeg -f concat -safe 0 -i input.txt -i static/songs/1.mp3 -vf "scale=1080:1920:force_original_aspect_ratio=decrease,pad=1080:1920:(ow-iw)/2:(oh-ih)/2" -c:v libx264 -c:a aac -shortest -r 30 -pix_fmt yuv420p output_reel_with_audio.mp4
